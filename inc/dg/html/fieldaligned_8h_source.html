408 template< class BinaryOp, class UnaryOp>
409 container FieldAligned<M,container>::evaluate( BinaryOp binary, UnaryOp unary, unsigned p0, unsigned rounds) const
410 {
411  //idea: simply apply I+/I- enough times on the init2d vector to get the result in each plane
412  //unary function is always such that the p0 plane is at x=0
413  assert( g_.Nz() > 1);
414  assert( p0 < g_.Nz());
415  const dg::Grid2d<double> g2d( g_.x0(), g_.x1(), g_.y0(), g_.y1(), g_.n(), g_.Nx(), g_.Ny());
416  container init2d = dg::evaluate( binary, g2d), temp(init2d), tempP(init2d), tempM(init2d);
417  container vec3d = dg::evaluate( dg::zero, g_);
418  std::vector<container> plus2d( g_.Nz(), (container)dg::evaluate(dg::zero, g2d) ), minus2d( plus2d), result( plus2d);
419  unsigned turns = rounds;
420  if( turns ==0) turns++;
421  //first apply Interpolation many times, scale and store results
422  for( unsigned r=0; r<turns; r++)
423  for( unsigned i0=0; i0<g_.Nz(); i0++)
424  {
425  tempP = init2d;
426  tempM = init2d;
427  unsigned rep = i0 + r*g_.Nz();
428  for(unsigned k=0; k<rep; k++)
429  {
430  dg::blas2::symv( plus, tempP, temp);
431  temp.swap( tempP);
432  dg::blas2::symv( minus, tempM, temp);
433  temp.swap( tempM);
434  }
435  dg::blas1::scal( tempP, unary( (double)rep*g_.hz() ) );
436  dg::blas1::scal( tempM, unary( -(double)rep*g_.hz() ) );
437  dg::blas1::axpby( 1., tempP, 1., plus2d[i0]);
438  dg::blas1::axpby( 1., tempM, 1., minus2d[i0]);
439  }
440  //now we have the plus and the minus filaments
441  if( rounds == 0) //there is a limiter
442  {
443  for( unsigned i0=0; i0<g_.Nz(); i0++)
444  {
445  int idx = (int)i0 - (int)p0;
446  if(idx>=0)
447  result[i0] = plus2d[idx];
448  else
449  result[i0] = minus2d[abs(idx)];
450  thrust::copy( result[i0].begin(), result[i0].end(), vec3d.begin() + i0*g2d.size());
451  }
452  }
453  else //sum up plus2d and minus2d
454  {
455  for( unsigned i0=0; i0<g_.Nz(); i0++)
456  {
457  unsigned revi0 = (g_.Nz() - i0)%g_.Nz(); //reverted index
458  dg::blas1::axpby( 1., plus2d[i0], 0., result[i0]);
459  dg::blas1::axpby( 1., minus2d[revi0], 1., result[i0]);
460  }
461  dg::blas1::axpby( -1., init2d, 1., result[0]);
462  for(unsigned i0=0; i0<g_.Nz(); i0++)
463  {
464  int idx = ((int)i0 -(int)p0 + g_.Nz())%g_.Nz(); //shift index
465  thrust::copy( result[idx].begin(), result[idx].end(), vec3d.begin() + i0*g2d.size());
466  }
467  }
468  return vec3d;
