27 CooSparseBlockMat save_outer_values(EllSparseBlockMat& m)
28 {
29  //search outer values in m
30  CooSparseBlockMat mat( m.num_rows, 2, m.n, m.left, m.right);
31  int index = m.data.size()/ m.n/m.n;
32  thrust::host_vector<double> data_element(m.n*m.n, 0), zero(data_element);
33  bool found=false;
34  for( int i=0; i<m.num_rows; i++)
35  for( int d=0; d<m.blocks_per_line; d++)
36  {
37  if( m.cols_idx[i*m.blocks_per_line+d]==-1)
38  {
39  for( int j=0; j<m.n*m.n; j++)
40  data_element[j] = m.data[ m.data_idx[i*m.blocks_per_line+d]*m.n*m.n + j];
41  mat.add_value( i, 0, data_element);
42  m.data_idx[i*m.blocks_per_line+d] = index; //
43  m.cols_idx[i*m.blocks_per_line+d] = 0;
44  found=true;
45  }
46  if( m.cols_idx[i*m.blocks_per_line+d]==m.num_cols)
47  {
48  for( int j=0; j<m.n*m.n; j++)
49  data_element[j] = m.data[ m.data_idx[i*m.blocks_per_line+d]*m.n*m.n + j];
50  mat.add_value( i, 1, data_element);
51  m.data_idx[i*m.blocks_per_line+d] = index;
52  m.cols_idx[i*m.blocks_per_line+d] = m.num_cols-1;
53  found=true;
54  }
55  }
56  if(found)
57  m.data.insert( m.data.end(), zero.begin(), zero.end());
58  return mat;
59 }

70 EllSparseBlockMat distribute_rows( const EllSparseBlockMat& src, int coord, const int* howmany)
71 {
72  if( howmany[1] == 1)
73  {
74  EllSparseBlockMat temp(src);
75  temp.left = temp.left/howmany[0];
76  temp.right = temp.right/howmany[2];
77  return temp;
78  }
79  assert( src.num_rows == src.num_cols);
80  int chunk_size = src.num_rows/howmany[1];
81  EllSparseBlockMat temp(chunk_size, chunk_size, src.blocks_per_line, src.data.size()/(src.n*src.n), src.n);
82  temp.left = src.left/howmany[0];
83  temp.right = src.right/howmany[2];
84  //first copy data elements (even though not all might be needed it doesn't slow down things either)
85  for( unsigned i=0; i<src.data.size(); i++)
86  temp.data[i] = src.data[i];
87  //now grab the right chunk of cols and data indices
88  for( unsigned i=0; i<temp.cols_idx.size(); i++)
89  {
90  temp.data_idx[i] = src.data_idx[ coord*(chunk_size*src.blocks_per_line)+i];
91  temp.cols_idx[i] = src.cols_idx[ coord*(chunk_size*src.blocks_per_line)+i];
92  //data indices are correct but cols are still the global indices (remapping a bit clumsy)
93  //first in the zeroth line the col idx might be (global)num_cols - 1 -> map that to -1
94  if( coord==0 && i/src.blocks_per_line == 0 && temp.cols_idx[i] == src.num_cols-1) temp.cols_idx[i] = -1;
95  //second in the last line the col idx mighty be 0 -> map to (global)num_cols
96  if( coord==(howmany[1]-1)&& (int)i/src.blocks_per_line == temp.num_rows-1 && temp.cols_idx[i] == 0) temp.cols_idx[i] = src.num_cols;
97  //Elements are now in the range -1, 0, 1,..., (global)num_cols
98  //now shift this range to chunk range -1,..,chunk_size
99  temp.cols_idx[i] = (temp.cols_idx[i] - coord*chunk_size );
100  }
101  return temp;
102 }
