615 int MPIGrid2d::pidOf( double x, double y) const
616 {
617  int dims[2], periods[2], coords[2];
618  MPI_Cart_get( comm, 2, dims, periods, coords);
619  coords[0] = (unsigned)floor( (x-g.x0())/g.lx()*(double)dims[0] );
620  coords[1] = (unsigned)floor( (y-g.y0())/g.ly()*(double)dims[1] );
621  //if point lies on or over boundary of last cell shift into current cell (not so good for periodic boundaries)
622  coords[0]=(coords[0]==dims[0]) ? coords[0]-1 :coords[0];
623  coords[1]=(coords[1]==dims[1]) ? coords[1]-1 :coords[1];
624  int rank;
625  if( MPI_Cart_rank( comm, coords, &rank) == MPI_SUCCESS )
626  return rank;
627  else
628  return -1;
629 }
