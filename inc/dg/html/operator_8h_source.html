379 T lr_pivot( dg::Operator<T>& m, std::vector<unsigned>& p)
380 {
381  //from numerical recipes
382  T pivot, determinant=(T)1;
383  unsigned pivotzeile, numberOfSwaps=0;
384  const size_t n = m.size();
385  p.resize( n);
386  for( size_t j = 0; j < n; j++) //gehe Spalten /Diagonale durch
387  {
388  //compute upper matrix except for the diagonal element (the pivot)
389  for( size_t i = 0; i< j; i++)
390  for( size_t k=0; k<i; k++)
391  m(i,j)-=m(i,k)*m(k,j);
392  //compute canditates for pivot elements
393  for( size_t i = j; i< n; i++)
394  for( size_t k=0; k<j; k++)
395  m(i,j)-=m(i,k)*m(k,j);
396  //search for absolute maximum of pivot candidates
397  pivot = m(j,j);
398  pivotzeile = j;
399  for( size_t i = j+1; i < n; i++)
400  if( fabs( m(i,j)) > fabs(pivot))
401  {
402  pivot = m(i,j), pivotzeile = i;
403  }
404 
405  if( pivot!= (T)0 )
406  {
407  if( pivotzeile != j)
408  {
409  m.swap_lines( pivotzeile, j);
410  numberOfSwaps++;
411  }
412  p[j] = pivotzeile;
413  //divide all elements below the diagonal by the pivot to get the lower matrix
414  for( size_t i=j+1; i<n; i++)
415  m(i,j) /= pivot;
416  determinant*=m(j,j);
417 
418  }
419  else
420  throw Message( "Matrix is singular!!");
421  }
422  if( numberOfSwaps % 2 != 0)
423  determinant*=-1.;
424  return determinant;
425

437 void lr_solve( const dg::Operator<T>& lr, const std::vector<unsigned>& p, std::vector<T>& b)
438 {
439  assert(p.size() == lr.size() && p.size() == b.size());
440  const size_t n = p.size();
441  // Vorwärtseinsetzen
442  for( size_t i = 0; i<n; i++)
443  {
444  //mache Zeilentausch
445  std::swap( b[ p[i] ], b[i]);
446  for( size_t j = 0; j < i; j++)
447  b[i] -= lr(i,j)*b[j];
448  }
449  // Rückwärtseinsetzen
450  for( int i = n-1; i>=0; i--)
451  {
452  for( size_t j = i+1; j < n; j++)
453  b[i] -= lr(i,j)*b[j];
454  b[i] /= lr(i,i);
455  }
