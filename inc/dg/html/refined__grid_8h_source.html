42 thrust::host_vector<double> equidist_ref( unsigned add_x, unsigned node, unsigned n, unsigned N, dg::bc bcx, unsigned howmany)
43 {
44  assert( howmany <= N);
45  assert( node <= N);
46  if( node != 0 && node != N) assert( howmany <= node && howmany <= N-node);
47  if( add_x == 0 || howmany == 0)
48  {
49  thrust::host_vector<double> w_( n*N, 1);
50  return w_;
51  }
52  assert( node <= N);
53  //there are add_x+1 finer cells per refined cell ...
54  thrust::host_vector< double> left( n*N+n*add_x*howmany, 1), right(left);
55  for( unsigned i=0; i<(add_x+1)*howmany; i++)//the original cell and the additional ones
56  for( unsigned k=0; k<n; k++)
57  left[i*n+k] = add_x + 1;
58  //mirror left into right
59  for( unsigned i=0; i<right.size(); i++)
60  right[i] = left[ (left.size()-1)-i];
61  thrust::host_vector< double> both( n*N+2*n*add_x*howmany, 1);
62  for( unsigned i=0; i<left.size(); i++)
63  both[i] *= left[i];
64  for( unsigned i=0; i<right.size(); i++)
65  both[i+n*add_x*howmany] *= right[i];
66  if( node == 0 && bcx != dg::PER) { return left; }
67  else if( node == N && bcx != dg::PER) { return right; }
68  else if((node == N || node == 0) && bcx == dg::PER) { return both; }
69  else
70  {
71  thrust::host_vector<double> w_ = both;
72  //now shift indices so that refinement is around nodes
73  for( unsigned i=0; i<both.size(); i++)
74  w_[((howmany*add_x+node)*n+i)%both.size()] = both[i];
75  return w_;
76  }
77 }

159 thrust::host_vector<double> normalize_weights_and_compute_abscissas( const Grid1d& g, thrust::host_vector<double>& weights)
160 {
161  //normalize weights
162  unsigned Nx_new = weights.size()/g.n();
163  for( unsigned i=0;i<weights.size(); i++)
164  weights[i] *= (double)g.N()/(double)Nx_new;
165

167  boundaries[0] = g.x0();
168  for( unsigned i=0; i<Nx_new; i++)

172  {
173  abs[i*g.n()+j] = (boundaries[i+1]+boundaries[i])/2. +

627 cusp::coo_matrix<int, double, cusp::host_memory> projection( const dg::RefinedGrid2d& g_fine)
628 {
629  //form the adjoint
630  thrust::host_vector<double> w_f = dg::create::weights( g_fine);
631  thrust::host_vector<double> v_c = dg::create::inv_weights( g_fine.associated() );
632  cusp::coo_matrix<int, double, cusp::host_memory> Wf( w_f.size(), w_f.size(), w_f.size());
633  cusp::coo_matrix<int, double, cusp::host_memory> Vc( v_c.size(), v_c.size(), v_c.size());
634  for( int i =0; i<(int)w_f.size(); i++)
635  {
636  Wf.row_indices[i] = Wf.column_indices[i] = i;
637  Wf.values[i] = w_f[i]/g_fine.weightsX()[i]/g_fine.weightsY()[i];
638  }
639  for( int i =0; i<(int)v_c.size(); i++)
640  {
641  Vc.row_indices[i] = Vc.column_indices[i] = i;
642  Vc.values[i] = v_c[i];
643  }
644  cusp::coo_matrix<int, double, cusp::host_memory> temp = interpolation( g_fine), A;
645  cusp::transpose( temp, A);
646  cusp::multiply( A, Wf, temp);
647  cusp::multiply( Vc, temp, A);
648  A.sort_by_row_and_column();
649  return A;
650 }
