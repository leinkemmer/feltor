424 cusp::coo_matrix<int, double, cusp::host_memory> projection( const dg::RefinedGridX2d& g_fine)
425 {
426  //form the adjoint
427  thrust::host_vector<double> w_f = dg::create::weights( g_fine);
428  thrust::host_vector<double> v_c = dg::create::inv_weights( g_fine.associated() );
429  cusp::coo_matrix<int, double, cusp::host_memory> Wf( w_f.size(), w_f.size(), w_f.size());
430  cusp::coo_matrix<int, double, cusp::host_memory> Vc( v_c.size(), v_c.size(), v_c.size());
431  for( int i =0; i<(int)w_f.size(); i++)
432  {
433  Wf.row_indices[i] = Wf.column_indices[i] = i;
434  Wf.values[i] = w_f[i]/g_fine.weightsX()[i]/g_fine.weightsY()[i];
435  }
436  for( int i =0; i<(int)v_c.size(); i++)
437  {
438  Vc.row_indices[i] = Vc.column_indices[i] = i;
439  Vc.values[i] = v_c[i];
440  }
441  cusp::coo_matrix<int, double, cusp::host_memory> temp = interpolation( g_fine), A;
442  cusp::transpose( temp, A);
443  cusp::multiply( A, Wf, temp);
444  cusp::multiply( Vc, temp, A);
445  A.sort_by_row_and_column();
446  return A;
