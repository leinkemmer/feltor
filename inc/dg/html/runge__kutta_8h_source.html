578 int integrateRK(RHS& rhs, const Vector& begin, Vector& end, double T_max, double eps_abs )
579 {
580  RK_classic<s, Vector > rk( begin);
581  Vector old_end(begin), temp(begin);
582  end = begin;
583  if( T_max == 0) return 0;
584  double dt = T_max/10;
585  int NT = 10;
586  double error = 1e10;
587 
588  while( error > eps_abs && NT < pow( 2, 18) )
589  {
590  dt /= 2.;
591  NT *= 2;
592  end = begin;
593 
594  int i=0;
595  while (i<NT)
596  {
597  rk( rhs, end, temp, dt);
598  if( !rhs.monitor( temp ) ) //sanity check
599  {
600  #ifdef DG_DEBUG
601  std::cout << "---------Got sanity error -> choosing smaller step size and redo integration" << " NT "<<NT<<" dt "<<dt<< std::endl;
602  #endif
603  break;
604  }
605  end.swap( temp); //end is one step further
606  i++;
607  }
608  error = rhs.error( end, old_end);
609  old_end = end;
610 #ifdef DG_DEBUG
611 #ifdef MPI_VERSION

615 #endif //MPI

617 #endif //DG_DEBUG
618  }

621  {
622  std::cerr << "ATTENTION: Runge Kutta failed to converge. Error is NAN! "<<std::endl;
623  return -1;
624  }

626  {
