185 void EllSparseBlockMat<value_type>::symv(const thrust::host_vector<value_type>& x, thrust::host_vector<value_type>& y) const
186 {
187  assert( y.size() == (unsigned)num_rows*n*left_size*right_size);
188  assert( x.size() == (unsigned)num_cols*n*left_size*right_size);
189 
190  //simplest implementation
191  for( int s=0; s<left_size; s++)
192  for( int i=0; i<num_rows; i++)
193  for( int k=0; k<n; k++)
194  for( int j=right_range[0]; j<right_range[1]; j++)
195  {
196  int I = ((s*num_rows + i)*n+k)*right_size+j;
197  y[I] =0;
198  for( int d=0; d<blocks_per_line; d++)
199  for( int q=0; q<n; q++) //multiplication-loop
200  y[I] += data[ (data_idx[i*blocks_per_line+d]*n + k)*n+q]*
201  x[((s*num_cols + cols_idx[i*blocks_per_line+d])*n+q)*right_size+j];
202  }
